### 解法一

#### 时空复杂度

时间复杂度：O(n*n^2)

空间复杂度:   O(n^2)

```js
 */
var subsets = function (nums) {
    let ans = []
    for (let i = 0; i < nums.length; i++) {
        let length = ans.length
        for (let j = 0; j < length; j++) {
            let newChild = [...ans[j], nums[i]]
            ans.push(newChild)
        }
        ans.push([nums[i]])
    }
    ans.push([])
    return ans
};
```

### 解法二

#### 时空复杂度

时间复杂度：O(n* 2^n)

空间复杂度:   O(n)



```js
var subsets = function (nums) {
    let ans = []
    let n = nums.length
    for (let i = 0; i < (1 << n); ++i) {
        const t = []
        for (let j = 0; j < n; ++j) {
            if (i & (1 << j)) {
                t.push(nums[j])
            }
        }
        ans.push(t)
    }
    return ans
};
```
