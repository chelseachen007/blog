### 206.反转链表

#### 解法一

##### 时空复杂度

```js
var reverseList = function (head) {
    let [prev, curr] = [null, head]
    while (curr) {
        let temp = curr.next
        curr.next = prev
        prev = curr
        curr = temp
    }
    return prev
};
```

```js
var reverseList = function (head) {
```
